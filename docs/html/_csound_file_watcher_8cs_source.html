122 private static void EditorUpdate()
123 {
124 var startTime = Time.realtimeSinceStartup;
125 if (startTime > _lastUpdate + _timeBetweenUpdates)
126 lock (_actionsQueue)
127 {
128 if (_quitting)
129 _actionsQueue.Clear();
130
131 if (_executeActions)
132 while (_actionsQueue.Count > 0)
133 {
134 var action = _actionsQueue.Dequeue();
135 if (action == null)
136 continue;
137 //Debug.Log($"{startTime} fileWatcher: action!");
138 action();
139 }
140 _lastUpdate = Time.realtimeSinceStartup;
141 }
142 }

144 private static void StartWatching(string filePath)
145 {
146 if (string.IsNullOrWhiteSpace(filePath) || !File.Exists(filePath)) return;
147 // Debug.Log($"fileWatcher: START WATCHING {filePath}");
148 FileSystemWatcher watcher = new FileSystemWatcher();
149 watcher.Path = Path.GetDirectoryName(filePath);
150 watcher.Filter = Path.GetFileName(filePath);
151 watcher.NotifyFilter = NotifyFilters.LastWrite;
152 watcher.Changed += Watcher_Changed;
153 watcher.EnableRaisingEvents = true;
154 fswInstances.Add(watcher);
155 }
156
157 private static void Watcher_Changed(object sender, System.IO.FileSystemEventArgs e)
158 {
159 if (e.ChangeType == WatcherChangeTypes.Changed)
160 {
161 var fileChanged = e.FullPath.Replace(Path.DirectorySeparatorChar, Path.AltDirectorySeparatorChar);
162 //Debug.Log("fileWatcher: fileChanged! " + fileChanged);
163
164 if (!_lastFileChangeDict.ContainsKey(fileChanged)) return;
165
166 var lastChange = _lastFileChangeDict[fileChanged];
167 //Debug.Log($"fileWatcher: {fileChanged} last change was at {lastChange}");
168 //ignore duplicate calls detected by FileSystemWatcher on file save
169 if (DateTime.Now.Subtract(lastChange).TotalMilliseconds < 1000)
170 {
171 //Debug.Log($"fileWatcher: IGNORING CHANGE AT {DateTime.Now}");
172 return;
173 }
174
175 //Debug.Log($"fileWatcher: CHANGE! {e.Name} changed at {DateTime.Now}, last change was {lastChange}");
176 _lastFileChangeDict[fileChanged] = DateTime.Now;
177
178 var result = TestCsoundForErrors(fileChanged);
179 //Debug.Log(result != 0 ?
180 // $"fileWatcher: Heuston we have a problem... Disabling all CsoundUnity instances for file: {fileChanged}" :
181 // "<color=green>Csound file has no errors!</color>"
182 //);
183
184 //Debug.Log($"fileWatcher: CsoundUnity instances associated with this file: {_pathsCsdListDict[fileChanged].Count}");
185 var list = _pathsCsdListDict[fileChanged];
186 for (var i = 0; i < list.Count; i++)
187 {
188 var csound = list[i];
189 lock (_actionsQueue)
190 _actionsQueue.Enqueue(() =>
191 {
192 if (result != 0)
193 {
194 csound.enabled = false;
195 }
196 else
197 {
198 Debug.Log($"<color=green>[CsoundFileWatcher] Updating csd: {csound.csoundFileName} in GameObject: {csound.gameObject.name}</color>");
199 csound.enabled = true;
200 //file changed but guid stays the same
201 csound.SetCsd(csound.csoundFileGUID);
202 }
203 });
204 }
205 }

239 static void OnHierarchyChanged()
240 {
241 if (Application.isPlaying) return;
242
243 // Debug.Log("fileWatcher: OnHierarchyChanged");
244 foreach (var fsw in fswInstances)
245 {
246 fsw.Changed -= Watcher_Changed;
247 }
248 fswInstances.Clear();
249 FindInstancesAndStartWatching();
250 }

252 private static void FindInstancesAndStartWatching()
253 {
254 csoundInstances = (CsoundUnity[])Resources.FindObjectsOfTypeAll(typeof(CsoundUnity));//as CsoundUnity[];
255 _pathsCsdListDict.Clear();
256 _lastFileChangeDict.Clear();
257
258 // Debug.Log($"fileWatcher: found {csoundInstances.Length} instance(s) of csound");
259 foreach (var csd in csoundInstances)
260 {
261 // get csd file path from the CsoundUnity instance and check if it exists
262 var filePath = csd.GetFilePath();
263 // Debug.Log("fileWatcher: FILEPATH " + filePath);
264 if (!File.Exists(filePath)) continue;
265
266 if (TestCsoundForErrors(filePath) != 0)
267 {
268 Debug.LogError($"fileWatcher: Heuston we have a problem... CsoundUnity disabled for file: {filePath}");
269 csd.enabled = false;
270 }
271 else
272 {
273 //Debug.Log("<color=green>fileWatcher: Csound file has no errors!</color>");
274 var csdString = File.ReadAllText(filePath);
275 // check if the csdString in the asset file is different from the one we have serialised in CsoundUnity
276 if (!csd.csoundString.Equals(csdString))
277 {
278 Debug.Log($"<color=green>[CsoundFileWatcher] Updating csd: {csd.csoundFileName} in GameObject: {csd.gameObject.name}</color>");
279 // content changed but guid stays the same
280 csd.SetCsd(csd.csoundFileGUID);
281 }
282 csd.enabled = true;
283 }
284
285 //Debug.Log("fileWatcher: found a csd asset at path: " + filePath);
286 if (_pathsCsdListDict.ContainsKey(filePath))
287 {
288 // Debug.Log("fileWatcher: csd is already watched, adding the csound script to the list of CsoundUnity instances to update");
289 _pathsCsdListDict[filePath].Add(csd);
290 _lastFileChangeDict[filePath] = DateTime.Now;
291 }
292 else
293 {
294 // Debug.Log("fileWatcher: new csd, creating a list of attached CsoundUnity instances");
295 var list = new List<CsoundUnity> { csd };
296 _pathsCsdListDict.Add(filePath, list);
297 _lastFileChangeDict.Add(filePath, DateTime.Now);
298 StartWatching(filePath);
299 // Debug.Log($"fileWatcher: added {filePath} to fileWatch");
300 }
301 }
302 }
